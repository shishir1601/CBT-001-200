PROC 1 DSNAME USERCAT
CONTROL NOPROMPT
SET SYSOUTTRAP = 300
LISTCAT ENTRIES('&DSNAME') ALL
IF &USERCAT = USERCAT THEN +
LISTCAT ENTRIES('&DSNAME') ALL CATALOG('&DSNAME')
SET SYSOUTTRAP = 0
SET CATLINES = &SYSOUTLINE
SET N = 0
LOOP: SET N = &N + 1
IF &N > &CATLINES THEN END
SET CATMSG = &&SYSOUTLINE&N
REVCATCH &STR(&CATMSG)
GOTO LOOP
