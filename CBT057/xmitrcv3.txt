PROC 0 DEBUG
CONTROL NOLIST MSG NOFLUSH PROMPT
IF &DEBUG NE &STR() THEN +
  CONTROL LIST MSG NOFLUSH PROMPT CONLIST SYMLIST
SET &SAVEPREF = &STR(&SYSPREF)
PROFILE PREFIX(&SYSUID)
ISPEXEC CONTROL ERRORS RETURN
ISPEXEC DISPLAY PANEL(XMITRCV3)
IF &LASTCC = 8 THEN +
  DO
    IF &SAVEPREF EQ &STR() THEN +
      PROFILE NOPREFIX
    ELSE +
      IF &SAVEPREF EQ &STR( ) THEN +
        PROFILE NOPREFIX
      ELSE +
        PROFILE PREFIX(&SAVEPREF)
     EXIT
  END
IF &LOGDEL EQ &STR(YES) THEN +
  DO
    CONTROL NOMSG
    SET SYSOUTTRAP = 20
    PDS LOG.MISC FIX RESET
    SET SYSOUTTRAP = 0
    ISPEXEC CONTROL DISPLAY REFRESH
    CONTROL MSG
  END
ELSE +
  DO
    ISPEXEC BROWSE DATASET(LOG.MISC)
  END
IF &SAVEPREF EQ &STR() THEN +
  PROFILE NOPREFIX
ELSE +
  IF &SAVEPREF EQ &STR( ) THEN +
    PROFILE NOPREFIX
  ELSE +
    PROFILE PREFIX(&SAVEPREF)
