PROC 0  DEBUG
CONTROL NOLIST NOMSG NOFLUSH PROMPT
IF &DEBUG NE &STR() THEN +
  CONTROL LIST MSG NOFLUSH PROMPT CONLIST SYMLIST
FREE F(SLSSTLIB)
ALLOCATE DDNAME(SLSSTLIB) DATASET('TSO.SLSS.ISPTABL') SHR REUSE
SLSSMAIN: +
SET SLSS = &STR(MAIN)
ISPEXEC VPUT (SLSS DEBUG) SHARED
%SLSSENQ ENQTALL
IF &LASTCC ^= 0 THEN GOTO DUMMTLIB
ISPEXEC SELECT PGM(SLSSMAIN) PARM(&DEBUG)
DUMMTLIB: +
ISPEXEC TBOPEN SLSSPROF NOWRITE LIBRARY(ISPTABL)
IF &LASTCC ^= 0 THEN DO
  %SLSSPROF &SYSUID &DEBUG
END DO
ISPEXEC TBEND SLSSPROF
FREE F(SLSSTLIB)
EXIT
