PROC 0 DEBUG
CONTROL NOLIST MSG NOFLUSH PROMPT
IF &DEBUG NE &STR() THEN +
  CONTROL LIST MSG NOFLUSH PROMPT CONLIST SYMLIST
SET &SAVEPREF = &STR(&SYSPREF)
PROFILE PREFIX(&SYSUID)
ISPEXEC CONTROL ERRORS RETURN
REDISP1: +
ISPEXEC DISPLAY PANEL(XMITRCV1)
IF &LASTCC = 8 THEN +
  DO
    IF &SAVEPREF EQ &STR() THEN +
      PROFILE NOPREFIX
    ELSE +
      IF &SAVEPREF EQ &STR( ) THEN +
        PROFILE NOPREFIX
      ELSE +
        PROFILE PREFIX(&SAVEPREF)
     EXIT
  END
IF &RCV1 EQ &STR() AND +
   &RCV2 EQ &STR() AND +
   &RCV3 EQ &STR() AND +
   &RCV4 EQ &STR() AND +
   &RCV5 EQ &STR() AND +
   &RCV6 EQ &STR() AND +
   &RCV7 EQ &STR() AND +
   &RCV8 EQ &STR() THEN +
   GOTO REDISP1
TRANSMIT (&RCV1 &RCV2 &RCV3 &RCV4 &RCV5 &RCV6 &RCV7 &RCV8)
ISPEXEC CONTROL DISPLAY REFRESH
IF &SAVEPREF EQ &STR() THEN +
  PROFILE NOPREFIX
ELSE +
  IF &SAVEPREF EQ &STR( ) THEN +
    PROFILE NOPREFIX
  ELSE +
    PROFILE PREFIX(&SAVEPREF)
